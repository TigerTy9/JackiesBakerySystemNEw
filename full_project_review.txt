    with open(output_file, 'w', encoding='utf-8') as outfile:

            indent = ' ' * 4 * level
            outfile.write(f"{indent}{os.path.basename(root)}/\n")
            sub_indent = ' ' * 4 * (level + 1)

                if f not in EXCLUDE_FILES:

                outfile.write(f"\n\nFILE: {relative_path}\n")

SECRET_KEY = os.getenv("SECRET_KEY", "SUPER_SECRET_BAKERY_KEY") 
ALGORITHM = "HS256"
ACCESS_TOKEN_EXPIRE_HOURS = 8

oauth2_scheme = OAuth2PasswordBearer(tokenUrl="login")

# --- Utility Functions ---

def hash_password(password: str):
    """Encodes plain text password into a secure hash."""
    return pwd_context.hash(password)

def verify_password(plain_password, hashed_password):
    """Checks if the provided password matches the stored hash."""
    return pwd_context.verify(plain_password, hashed_password)

def create_access_token(data: dict):
    """Generates a JWT token for a validated user[cite: 89]."""
    to_encode = data.copy()
    expire = datetime.utcnow() + timedelta(hours=ACCESS_TOKEN_EXPIRE_HOURS)
    to_encode.update({"exp": expire})
    return jwt.encode(to_encode, SECRET_KEY, algorithm=ALGORITHM)

# --- Dependency Injection ---

def get_current_user(token: str = Depends(oauth2_scheme), db: Session = Depends(database.get_db)):
    """
    The security 'guard'. Extract this into your routes to identify 
    which bakery (tenant) is making the request[cite: 36, 66].
    """
    credentials_exception = HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="Could not validate credentials",
        headers={"WWW-Authenticate": "Bearer"},
    )
    try:
        payload = jwt.decode(token, SECRET_KEY, algorithms=[ALGORITHM])
        username: str = payload.get("sub")
        if username is None:
            raise credentials_exception
    except JWTError:
        raise credentials_exception
        
    user = db.query(models.User).filter(models.User.username == username).first()
    if user is None:
        raise credentials_exception
    return user

def check_admin_or_owner(current_user: models.User):
    if current_user.role not in ["admin", "owner"]:
        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail="Operation restricted to Owners/Admins"
        )
    return True

def get_tenant_db(
    current_user: models.User = Depends(get_current_user), 
    db: Session = Depends(database.get_db)
):
    """
    SaaS Security Layer: Injects the tenant_id into the PostgreSQL session.
    RLS will automatically filter all subsequent queries in this transaction.
    """
    # SET LOCAL only lasts for the duration of this specific database transaction
    db.execute(text(f"SET LOCAL app.tenant_id = '{current_user.tenant_id}'"))
    
    return db

# utils.py

def convert_to_base(quantity: float, from_unit: str, multiplier: float) -> float:
    """
    Converts any incoming purchase quantity to the base_unit (usually grams).
    Example: 1 Bag (from_unit) * 22679.6 (multiplier) = 22679.6g
    """
    return quantity * multiplier

def get_margin_status(fifo_margin: float, newest_margin: float) -> str:
    """
    Business logic to flag if the baker needs to raise prices.
    """
    if newest_margin < (fifo_margin * 0.9): # 10% drop in margin
        return "CRITICAL: Price Hike Detected"
    return "Healthy"

def get_conversion_multiplier(db, tenant_id: int, from_unit: str, to_unit: str):
    """Looks up the multiplier in the unit_conversions table."""
    conversion = db.query(models.UnitConversion).filter(
        models.UnitConversion.tenant_id == tenant_id,
        models.UnitConversion.from_unit == from_unit,
        models.UnitConversion.to_unit == to_unit
    ).first()
    return conversion.multiplier if conversion else 1.0

Building a fully integrated bakery management and point-of-sale platform is a highly rewarding project. Structuring it with Python and PostgreSQL provides a scalable, robust foundation, especially since your ultimate goal is to package and sell this software commercially. 

To make this a viable product for multiple businesses, it needs to be architected as a Software-as-a-Service application from day one. Here is a breakdown of the essential features and how to structure the backend to handle your specific inventory and margin requirements.

## Essential System Features

* **Bill of Materials Management:** A recipe engine that maps finished baked goods to exact quantities of raw ingredients.
* **Lot-Based Inventory Tracking:** To calculate margins based on oldest versus newest ingredient costs, the system must track individual purchase batches rather than just a running total of ingredients.
* **Automated Deductions:** A trigger system that automatically deducts raw ingredient quantities from the database the moment a finished product is logged as baked or sold.
* **Dynamic Margin Analysis:** A dashboard displaying profit margins per item, calculated dynamically against both the First-In-First-Out cost and the most recent replacement cost.
* **Immutable Transaction Logs:** A highly secure ledger recording all sales, inventory adjustments, and waste, ensuring accurate accounting and audit trails.
* **Multi-Tenancy Architecture:** Data isolation so multiple independent bakeries can use the software simultaneously without their data intersecting.



## Database Design Strategy in PostgreSQL

To achieve the precise margin calculations you want, your database schema needs to separate the concept of an "Ingredient" from an "Ingredient Purchase". 

* **Products Table:** Stores the finished items ready for sale (e.g., Sourdough Loaf).
* **Ingredients Table:** The master list of raw materials (e.g., Bread Flour, Salt).
* **Ingredient_Lots Table:** This is the critical piece. Every time flour is purchased, a new row is created here logging the date, quantity, and specific price paid.
* **Recipes Table:** Maps Products to Ingredients, specifying the exact weight or volume needed.
* **Transactions Table:** Logs every sale, linking the Product, sale price, and timestamp.

## Effective Coding Implementation in Python

For a clean, maintainable backend, a framework like Django or FastAPI is highly recommended. Django offers a built-in Object-Relational Mapper that handles complex PostgreSQL queries beautifully, while FastAPI is exceptionally fast and excellent for building modern web application programming interfaces.

* **Handling the Auto-Subtract Logic:** When a user clicks "Bake" on a batch of cookies, your Python code should initiate a database transaction. It will look up the recipe, calculate the total raw ingredients needed, and deduct those quantities from the `Ingredient_Lots` table, starting with the oldest available lot first to maintain accurate physical inventory.
* **Calculating the Margins:** To show the margin on the "oldest item", your Python logic queries the oldest active lot for each ingredient in the recipe and sums their costs. For the "newest purchase" margin, it queries the most recently added lot for those same ingredients. Subtracting these costs from the final retail price gives the baker a clear view of their true profit spread.
* **Ensuring Data Integrity:** By wrapping the deduction logic in standard database transactions, you ensure that if an error occurs halfway through deducting ingredients, the entire operation rolls back. This prevents corrupted inventory counts.

Building this with a focus on strict data validation and secure, isolated environments will make the transition from a single-bakery tool to a commercial software product much smoother.

Yes, looking at the full scope of building a commercial, production-ready SaaS for bakeries, there are several crucial business and technical layers missing from that initial outline. 

If you want this platform to be genuinely useful for a growing baking business and marketable to others, here are the critical missing components:

### The Bakery Business Logic

* **Complex Unit Conversions:** This is the most common point of failure in culinary software. A baker buys flour in 50-pound bags, stores it in large bins, but writes recipes in grams. Your database needs a robust conversion engine so a purchase of "1x 50lb bag" accurately translates to "22,679 grams" in the available inventory pool.
* **Packaging and Consumables:** A cake is not just flour and sugar. Boxes, cake boards, custom labels, and parchment paper all cost money. To get true profit margins, your Bill of Materials must include these non-edible inventory items.
* **Spoilage and Waste Tracking:** Ingredients expire, and baked goods go stale or get dropped. If waste is not tracked, the digital inventory will permanently drift away from the physical reality, ruining the margin calculations. 
* **Custom Orders and Workflow States:** Many independent bakers do a mix of standard retail sales and custom orders (like wedding cakes). The system needs a pipeline for these custom orders: Quote Pending, Deposit Paid, Baking Scheduled, and Ready for Pickup.

### The SaaS Technical Architecture

* **Row-Level Security (RLS) in PostgreSQL:** Since you plan to sell this software, multi-tenancy is critical. If multiple bakeries are on the same database, you must guarantee their data never leaks. Implementing RLS directly in PostgreSQL ensures that even if there is an application-level bug, a user from Bakery A cannot physically query data belonging to Bakery B.
* **Role-Based Access Control (RBAC):** The business owner needs to see financial margins and system configurations, but a front-of-house cashier or a kitchen assistant should only see the Point of Sale or the daily baking queue. 
* **Overhead and Labor Apportionment:** For the margin analysis to be perfectly accurate, the system should eventually allow the user to input hourly labor costs and fixed overhead (like web domain hosting, email services, or kitchen rental) to calculate the true Net Margin, not just the Gross Margin of the ingredients.
* **Audit Logging for Security:** You mentioned transaction logs, but a commercial system needs full audit trails for system events. If an inventory count is manually adjusted or a custom order invoice is deleted, the system needs to record who did it and when, providing a secure, tamper-proof history.

def login_for_access_token(

    # 1. Look up the user

from sqlalchemy.orm import Session
from sqlalchemy import desc
import models, schemas

def process_bake_and_calculate_margins(db: Session, product_id: int, tenant_id: int, quantity_sold: int = 1):
    """
    Processes a bake/sale by deducting raw ingredients from lots using FIFO logic.
    Calculates margins based on both the oldest and newest purchase costs.
    """
    product = db.query(models.Product).filter(models.Product.id == product_id).first()
    if not product:
        raise Exception("Product not found")
    
    total_cost_fifo = 0.0
    total_cost_newest = 0.0
    
    # 1. Loop through every ingredient in the recipe 
    for item in product.recipe_items:
        # --- BULK CALCULATION ---
        # Multiply the single recipe requirement by the number of items actually baked
        total_qty_needed = item.quantity_required * quantity_sold
        
        # --- NEWEST PURCHASE PRICE CALCULATION ---
        # We look up the most recent lot to see what it would cost to replace these ingredients today
        newest_lot = db.query(models.IngredientLot)\
            .filter(models.IngredientLot.ingredient_id == item.ingredient_id)\
            .order_by(desc(models.IngredientLot.purchase_date)).first()
        
        if newest_lot:
            cost_per_unit_newest = newest_lot.cost_total / newest_lot.quantity_purchased
            total_cost_newest += (cost_per_unit_newest * total_qty_needed)

        # --- FIFO DEDUCTION & OLDEST PRICE CALCULATION --- 
        # We deduct from the oldest available lots first until the requirement is met
        remaining_to_deduct = total_qty_needed
        
        while remaining_to_deduct > 0:
            print(f"DEBUG: Ingredient ID {item.ingredient_id} - Need: {remaining_to_deduct}")
            
            oldest_lot = db.query(models.IngredientLot)\
                .filter(models.IngredientLot.ingredient_id == item.ingredient_id, 
                        models.IngredientLot.is_depleted == False)\
                .order_by(models.IngredientLot.purchase_date).first()
            
            if not oldest_lot:
                print("DEBUG: Out of stock, breaking loop")
                raise Exception(f"Out of stock: {item.ingredient.name}")

            cost_per_unit_fifo = oldest_lot.cost_total / oldest_lot.quantity_purchased
            
            if oldest_lot.quantity_remaining >= remaining_to_deduct:
                oldest_lot.quantity_remaining -= remaining_to_deduct
                total_cost_fifo += (cost_per_unit_fifo * remaining_to_deduct)
                remaining_to_deduct = 0 
            else:
                total_cost_fifo += (cost_per_unit_fifo * oldest_lot.quantity_remaining)
                remaining_to_deduct -= oldest_lot.quantity_remaining
                oldest_lot.quantity_remaining = 0
                oldest_lot.is_depleted = True
            
            # FOR TESTING: This prevents the 'hang'
            db.flush()
        
    # 2. Record the transaction with the margins 
    # Note: retail_price is multiplied by quantity_sold for accurate totals
    total_revenue = product.retail_price * quantity_sold
    margin_fifo = total_revenue - total_cost_fifo
    margin_newest = total_revenue - total_cost_newest
    
    new_log = models.TransactionLog(
        tenant_id=tenant_id,
        product_id=product_id,
        sale_price=total_revenue, # Total for the whole batch
        margin_fifo=margin_fifo,
        margin_newest=margin_newest
    )
    
    db.add(new_log)
    db.commit()
    return new_log

    # LOOK MA, NO FILTERS! 
    # Because of RLS, Postgres will ONLY return ingredients for this specific tenant.
    # If a developer forgets the filter, the data is still perfectly safe.
    ingredients = db.query(models.Ingredient).all()
    
    return ingredients

@router.get("/stock-levels", response_model=List[schemas.IngredientStockResponse])
def get_stock_levels(db: Session = Depends(get_tenant_db)):
    """
    Calculates the total quantity on hand for every ingredient 
    by summing up all non-depleted lots.
    """
    # We join Ingredients and IngredientLots to get names and quantities together
    stock = db.query(
        models.Ingredient.id.label("ingredient_id"),
        models.Ingredient.name,
        func.sum(models.IngredientLot.quantity_remaining).label("total_quantity"),
        models.Ingredient.base_unit
    ).join(models.IngredientLot).filter(
        models.IngredientLot.is_depleted == False
    ).group_by(models.Ingredient.id).all()
    
    return stock

document.getElementById('login-form').addEventListener('submit', async (e) => {
    e.preventDefault();
    const formData = new URLSearchParams();
    formData.append('username', document.getElementById('username').value);
    formData.append('password', document.getElementById('password').value);

    try {
        const response = await fetch(`${API_URL}/login`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
            body: formData
        });

        if (!response.ok) throw new Error("Invalid credentials");
        const data = await response.json();
        localStorage.setItem('bakery_token', data.access_token);
        
        document.getElementById('login-error').classList.add('hidden');
        showDashboard();
    } catch (err) {
        const errorEl = document.getElementById('login-error');
        errorEl.textContent = err.message;
        errorEl.classList.remove('hidden');
    }
});

function logout() {
    localStorage.removeItem('bakery_token');
    document.getElementById('dashboard-view').classList.add('hidden');
    document.getElementById('login-view').classList.remove('hidden');
    document.getElementById('login-view').classList.add('flex');
}

        alert("Order Quote Drafted!");
        e.target.reset();
        document.getElementById('order-item-rows').innerHTML = '';
        addOrderItemRow();
        fetchOrders();
    } catch (err) {
        alert(err.detail || "Failed to create order");
    }
});

async function fetchOrders() {
    try {
        const res = await fetch(`${API_URL}/orders/pipeline`, { headers: getHeaders() });
        if (res.status === 401) return logout();
        const orders = await res.json();
        
        const tbody = document.getElementById('orders-table-body');
        tbody.innerHTML = orders.map(order => `
            <tr>
                <td class="p-2 border-b font-medium">#${order.id}</td>
                <td class="p-2 border-b">${order.customer_name}</td>
                <td class="p-2 border-b">${new Date(order.delivery_date).toLocaleDateString()}</td>
                <td class="p-2 border-b">
                    <span class="px-2 py-1 bg-yellow-100 text-yellow-800 text-xs rounded-full font-bold">
                        ${order.status}
                    </span>
                </td>
            </tr>
        `).join('');
    } catch (err) { console.error(err); }
}

        async function loginAsUser(id) {
            if(!confirm("Are you sure you want to log in as this user? You will be redirected to their dashboard.")) return;
            
            try {
                const res = await fetch(`${API_URL}/admin/impersonate/${id}`, { method: 'POST' });
                if(!res.ok) throw new Error("Failed to generate token");
                
                const data = await res.json();
                
                // 1. Overwrite your local token with this user's newly generated token
                localStorage.setItem('bakery_token', data.access_token);
                
                // 2. Redirect your browser straight to the main app dashboard
                window.location.href = 'index.html'; 
            } catch (err) { 
                console.error(err); 
                alert("Could not log in as user.");
            }
        }
